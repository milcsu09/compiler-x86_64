struct A {
  a : i64;
  b : i64;
}

struct B {
  a1 : struct A;
  a2 : struct A;
}

fn B_init(b : *struct B) {
  b->a1.a = 10;
  b->a1.b = 20;

  b->a2.a = 30;
  b->a2.b = 40;
}

fn main() -> i32 {
  b : struct B;

  B_init(&b);

  print b.a1.a;
  print b.a1.b;

  print b.a2.a;
  print b.a2.b;

  return 0;
}


fn square (n : i64) i64
{
  return n * n;
}

fn map (a : *i64, i : i64, f : fn (i64) i64) void
{
  for ; i >= 0; i = i - 1
    {
      a[i] = f (a[i]);
    }
}

fn main () i32
{
  a : [10]i64;
  i : i64;

  for i = 0; i < 10; i = i + 1
    {
      a[i] = i + 1;
    }

  map (a, 10, square);

  for i = 0; i < 10; i = i + 1
    {
      print a[i];
    }

  return 0;
}

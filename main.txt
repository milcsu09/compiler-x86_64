
fn is_even(n : i64) u8 {
  // No % yet.
  return n - 2 * (n / 2) == 0;
}


fn main() i32 {
  // https://projecteuler.net/problem=2
  a : i64;
  b : i64;

  a = 1;
  b = 2;

  s : i64;
  s = 0;

  while a < 4000000 {
    if is_even(a) {
      s = s + a;
    }

    c : i64;
    c = a + b;

    a = b;
    b = c;
  }

  print s;

  return 0;
}

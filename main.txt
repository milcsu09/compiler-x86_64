
fn identity (n : i64) i64
{
  return n;
}

fn successor (n : i64) i64
{
  return n + 1;
}

fn double (n : i64) i64
{
  return n * 2;
}

fn square (n : i64) i64
{
  return n * n;
}

fn main () i32
{
  fs : [4]fn (i64) i64;

  fs[0] = identity;
  fs[1] = successor;
  fs[2] = double;
  fs[3] = square;

  N : i64;
  N = 5;

  i : i64;

  for i = 0; i < 4; i = i + 1
    {
      print fs[i] (N);
    }

  return 0;
}

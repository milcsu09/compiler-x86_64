struct S {
  a : i8;
  b : i16;
  c : i32;
  d : i64;
  e : [4]u8;
}

fn main() -> i32 {
  s : struct S;

  s.a = 10;
  s.b = 20;
  s.c = 30;
  s.d = 40;

  s.e[0] = 1;
  s.e[1] = 2;
  s.e[2] = 3;
  s.e[3] = 4;

  print s.a;
  print s.b;
  print s.c;
  print s.d;

  print s.e[0];
  print s.e[1];
  print s.e[2];
  print s.e[3];

  return 0;
}

struct a {
  x : i32;
  y : i32;
}

struct b {
  pa : *struct a;
}

struct c {
  pb : *struct b;
}

fn main() -> i32 {
  a : struct a;
  b : struct b;
  c : struct c;

  b.pa = &a;
  c.pb = &b;

  p : *struct c;
  p = &c;

  p->pb->pa->x = 10;
  p->pb->pa->y = 20;

  print a.x;
  print a.y;

  return 0;
}
